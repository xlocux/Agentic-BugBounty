+ Nuovo: scripts/lib/session.js
Tutta la logica session in un modulo separato — run-pipeline.js lo chiama, non lo conosce nei dettagli.
// Funzioni esportate:

~ Modifiche run-pipeline.js (3 punti chirurgici)
// 1. Prima di avviare — asset selection checkpoint
- const assetsToAnalyze = config.additional_assets || [];
+ if (isHitlMode(args)) {

// 2. Dopo explorer, prima di researcher — plan approval
- const domains = ALL_DOMAINS[config.asset_type];
+ let domains = ALL_DOMAINS[config.asset_type];
+ if (isHitlMode(args)) {
+ const plan = buildPlanForAssetType(config.asset_type, surfaceMap);
+ writeCheckpoint(sessionPath, { status:"awaiting_approval", phase:"researcher", ...plan });

+ domains = domains.filter(d => resp.approved_ops.includes(d.id));
+ }
// 3. Dopo researcher, prima di triage — findings review
